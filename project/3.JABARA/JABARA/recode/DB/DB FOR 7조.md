CREATE VIEW RANKING AS
select p.USER_ID,  ROW_NUMBER() over(order by count(p.POST_NUMBER) desc) as `RANK`
from POST p
LEFT OUTER JOIN POST q
on (p.POST_CATEGORY = q.POST_CATEGORY)
where p.POST_CATEGORY = 'CODE'
group by p.USER_ID;
